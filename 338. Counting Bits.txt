338. Counting Bits

Explanation:
DP: number of 1s in i = number of 1s in i/2 + last bit of i.

class Solution {
    public int[] countBits(int n) {
        int[] res = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            res[i] = res[i >> 1] + (i & 1);
        }
        return res;
    }
}
