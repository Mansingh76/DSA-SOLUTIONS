684. Redundant Connection


class Solution {
    public int[] findRedundantConnection(int[][] edges) {
        int n = edges.length;
        int[] parent = new int[n + 1];
        int[] rank = new int[n + 1]; // for union by rank

        // Initialize parent pointers
        for (int i = 1; i <= n; i++) {
            parent[i] = i;
        }

        // Process edges
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1];
            if (!union(u, v, parent, rank)) {
                // If union fails, u and v are already connected â†’ cycle edge
                return edge;
            }
        }
        return new int[0]; // should not happen
    }

    // Find with path compression
    private int find(int x, int[] parent) {
        if (x != parent[x]) {
            parent[x] = find(parent[x], parent); 
        }
        return parent[x];
    }

    // Union by rank
    private boolean union(int x, int y, int[] parent, int[] rank) {
        int rootX = find(x, parent);
        int rootY = find(y, parent);

        if (rootX == rootY) return false; // already connected (cycle)

        if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
        return true;
    }
}
