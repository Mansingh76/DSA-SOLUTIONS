
567. Permutation in String



class Solution {
    public boolean checkInclusion(String s1, String s2) {
        if (s1.length() > s2.length()) return false;

        // Frequency count arrays for characters 'a' to 'z'
        int[] s1Count = new int[26];
        int[] s2Count = new int[26];

        // Initialize frequency of first s1.length() characters
        for (int i = 0; i < s1.length(); i++) {
            s1Count[s1.charAt(i) - 'a']++;
            s2Count[s2.charAt(i) - 'a']++;
        }

        // If first window matches, return true
        if (matches(s1Count, s2Count)) return true;

        // Slide the window over s2
        for (int i = s1.length(); i < s2.length(); i++) {
            // Add new character (right side of window)
            s2Count[s2.charAt(i) - 'a']++;

            // Remove old character (left side of window)
            s2Count[s2.charAt(i - s1.length()) - 'a']--;

            // Check if current window matches
            if (matches(s1Count, s2Count)) return true;
        }

        return false;
    }

    // Helper to check if two frequency arrays are identical
    private boolean matches(int[] a, int[] b) {
        for (int i = 0; i < 26; i++) {
            if (a[i] != b[i]) return false;
        }
        return true;
    }
}
